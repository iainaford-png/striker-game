    <h1>Striker-like: Lower League Hero</h1>

    <div class="sub">Stop the pointer to shoot. Climb divisions as the pointer gets faster.</div>

    <canvas id="c" width="760" height="420"></canvas>

    <div class="row">

      <div class="card">

        <div class="stat">Division</div>

        <div class="big"><span id="division">4</span> <span class="pill" id="speedPill">Slow</span></div>

        <div class="stat" style="margin-top:10px;">Season record (this division)</div>

        <div class="big"><span id="record">0–0</span></div>

        <div class="hint">Win matches to earn points. Top spots promote, bottom spots relegate.</div>

      </div>

      <div class="card">

        <div class="stat">Match</div>

        <div class="big">Fixture <span id="fixture">1</span>/10</div>

        <div class="stat" style="margin-top:10px;">Shots remaining</div>

        <div class="big"><span id="shots">5</span></div>

        <div class="msg" id="message">&nbsp;</div>

        <div class="hint">Controls: <b>Space</b> / <b>Enter</b> or tap <b>SHOOT</b>.</div>

      </div>

        <button class="primary" id="shootBtn" style="width:100%; margin-bottom:10px;">SHOOT</button>

        <button id="nextBtn" style="width:100%;">NEXT FIXTURE</button>

<script>

(() => {

  // ---------- Game Design ----------

  // Divisions: 4 (slow) -> 1 (fast)

  // Each fixture: 5 shots. Need enough goals to "win" the match.

  // Simple season (10 fixtures) per division. Promotion/relegation based on points.

  //

  // You can tweak these constants to taste.

  const FIXTURES_PER_SEASON = 10;

  const SHOTS_PER_FIXTURE = 5;

  // How many goals count as a win/draw in a fixture (based on your goals vs AI goals)

  // AI goals loosely scale with division.

  function aiGoalsForDivision(div) {

    // div 4 easiest, div 1 hardest

    const base = {4:1, 3:1, 2:2, 1:2}[div];

    // small randomness

    return base + (Math.random() < 0.35 ? 1 : 0) - (Math.random() < 0.20 ? 1 : 0);

  }

  // Pointer speed increases as division gets higher.

  function speedForDivision(div) {

    // pixels per frame-ish (we'll multiply by delta)

    return {4: 180, 3: 240, 2: 320, 1: 420}[div];

  }

  // Goal window (how close to the center counts as a goal) shrinks a bit with difficulty.

  function goalWindowForDivision(div) {

    return {4: 64, 3: 56, 2: 48, 1: 42}[div];

  }

  function speedLabel(div) {

    return {4:"Slow", 3:"Normal", 2:"Fast", 1:"Rapid"}[div];

  }

  // Promotion/relegation thresholds over 10 fixtures

  // 3 pts win, 1 pt draw

  const PROMOTE_AT_OR_ABOVE = 22; // usually ~7 wins + draw

  const RELEGATE_AT_OR_BELOW = 10; // usually <= 3 wins+draws

  // ---------- State ----------

  let division = 4;

  let fixture = 1;

  let shotsLeft = SHOTS_PER_FIXTURE;

  let goalsThisFixture = 0;

  let aiGoals = aiGoalsForDivision(division);

  let points = 0;

  let wins = 0;

  let losses = 0;

  let draws = 0;

  let canShoot = true;

  let fixtureOver = false;

  // Pointer movement

  let pointerX = 0;

  let pointerDir = 1; // 1 right, -1 left

  let lastT = performance.now();

  // ---------- DOM ----------

  const canvas = document.getElementById("c");

  const ctx = canvas.getContext("2d");

  const divisionEl = document.getElementById("division");

  const speedPill = document.getElementById("speedPill");

  const fixtureEl = document.getElementById("fixture");

  const shotsEl = document.getElementById("shots");

  const recordEl = document.getElementById("record");

  const msgEl = document.getElementById("message");

  const shootBtn = document.getElementById("shootBtn");

  const nextBtn = document.getElementById("nextBtn");

  function setMessage(text, kind="") {

    msgEl.textContent = text || "\u00A0";

    msgEl.className = "msg" + (kind ? " " + kind : "");

  }

  function updateHUD() {

    divisionEl.textContent = division;

    speedPill.textContent = speedLabel(division);

    fixtureEl.textContent = fixture;

    shotsEl.textContent = shotsLeft;

    recordEl.textContent = `${wins}-${losses}` + (draws ? `-${draws}` : "");

  }

  // ---------- Match flow ----------

  function resetFixture() {

    shotsLeft = SHOTS_PER_FIXTURE;

    goalsThisFixture = 0;

    aiGoals = aiGoalsForDivision(division);

    canShoot = true;

    fixtureOver = false;

    setMessage("Aim and shoot!", "");

    pointerX = 0;

    pointerDir = 1;

    updateHUD();

  }

  function finishFixture() {

    fixtureOver = true;

    canShoot = false;

    let result;

    if (goalsThisFixture > aiGoals) {

      points += 3; wins++;

      result = `WIN ${goalsThisFixture}-${aiGoals} (+3 pts)`;

      setMessage(result, "good");

    } else if (goalsThisFixture === aiGoals) {

      points += 1; draws++;

      result = `DRAW ${goalsThisFixture}-${aiGoals} (+1 pt)`;

      setMessage(result, "");

    } else {

      losses++;

      result = `LOSS ${goalsThisFixture}-${aiGoals} (+0 pts)`;

      setMessage(result, "bad");

    }

    updateHUD();

  }

  function finishSeason() {

    // Promotion / Relegation rules (except boundaries)

    let seasonMsg = `Season finished on ${points} pts. `;

    if (division > 1 && points >= PROMOTE_AT_OR_ABOVE) {

      division -= 1;

      seasonMsg += `PROMOTED to Division ${division}!`;

      setMessage(seasonMsg, "good");

    } else if (division < 4 && points <= RELEGATE_AT_OR_BELOW) {

      division += 1;

      seasonMsg += `RELEGATED to Division ${division}.`;

      setMessage(seasonMsg, "bad");

    } else {

      seasonMsg += `You stay in Division ${division}.`;

      setMessage(seasonMsg, "");

    }

    // Reset season stats

    points = 0; wins = 0; losses = 0; draws = 0;

    fixture = 1;

    resetFixture();

  }

  function nextFixture() {

    if (!fixtureOver) return;

    fixture++;

    if (fixture > FIXTURES_PER_SEASON) {

      finishSeason();

      return;

    }

    resetFixture();

  }

  // ---------- Shooting logic ----------

  function attemptShot() {

    if (!canShoot || fixtureOver) return;

    canShoot = false;

    const goalCenterX = canvas.width * 0.5;

    const dist = Math.abs(pointerX - goalCenterX);

    const window = goalWindowForDivision(division);

    // "Stop pointer": freeze briefly, evaluate, then resume (unless out of shots)

    const isGoal = dist <= window;

    shotsLeft--;

    if (isGoal) {

      goalsThisFixture++;

      setMessage(`GOAL! (${goalsThisFixture})`, "good");

    } else {

      setMessage("Saved / Missed!", "bad");

    }

    updateHUD();

    // End fixture if no shots left

    if (shotsLeft <= 0) {

      finishFixture();

      return;

    }

    // brief pause then re-enable shooting

    setTimeout(() => {

      canShoot = true;

      setMessage("Next shot…", "");

    }, 450);

  }

  // ---------- Input ----------

  shootBtn.addEventListener("click", attemptShot);

  nextBtn.addEventListener("click", nextFixture);

  window.addEventListener("keydown", (e) => {

    if (e.code === "Space" || e.code === "Enter") {

      e.preventDefault();

      attemptShot();

    }

    if (e.code === "KeyN") nextFixture();

  });

  // ---------- Render ----------

  function drawPitch() {

    const w = canvas.width, h = canvas.height;

    // background

    ctx.clearRect(0,0,w,h);

    // subtle pitch bands

    for (let i=0; i<8; i++) {

      ctx.fillStyle = i % 2 === 0 ? "rgba(80, 160, 120, 0.15)" : "rgba(80, 160, 120, 0.10)";

      ctx.fillRect(0, i*(h/8), w, h/8);

    }

    // goal area (top)

    const goalW = w * 0.68;

    const goalH = h * 0.18;

    const goalX = (w - goalW)/2;

    const goalY = h * 0.08;

    // goal frame

    ctx.lineWidth = 4;

    ctx.strokeStyle = "rgba(231,240,255,0.9)";

    ctx.strokeRect(goalX, goalY, goalW, goalH);

    // net lines

    ctx.lineWidth = 1;

    ctx.strokeStyle = "rgba(231,240,255,0.20)";

    for (let x = goalX; x <= goalX+goalW; x += 18) {

      ctx.beginPath(); ctx.moveTo(x, goalY); ctx.lineTo(x, goalY+goalH); ctx.stroke();

    }

    for (let y = goalY; y <= goalY+goalH; y += 14) {

      ctx.beginPath(); ctx.moveTo(goalX, y); ctx.lineTo(goalX+goalW, y); ctx.stroke();

    }

    // target "goal window"

    const centerX = w * 0.5;

    const window = goalWindowForDivision(division);

    ctx.fillStyle = "rgba(79,209,197,0.18)";

    ctx.fillRect(centerX - window, goalY, window*2, goalH);

    ctx.strokeStyle = "rgba(79,209,197,0.65)";

    ctx.lineWidth = 2;

    ctx.strokeRect(centerX - window, goalY, window*2, goalH);

    // penalty spot

    ctx.fillStyle = "rgba(231,240,255,0.8)";

    ctx.beginPath();

    ctx.arc(centerX, h*0.62, 4, 0, Math.PI*2);

    ctx.fill();

    // HUD text inside canvas

    ctx.fillStyle = "rgba(231,240,255,0.85)";

    ctx.font = "bold 16px system-ui, -apple-system, Segoe UI, Roboto, Arial";

    ctx.fillText(`Division ${division} • Fixture ${fixture}/${FIXTURES_PER_SEASON}`, 16, 28);

    ctx.font = "14px system-ui, -apple-system, Segoe UI, Roboto, Arial";

    ctx.fillStyle = "rgba(159,179,209,0.95)";

    ctx.fillText(`Goals: ${goalsThisFixture}  |  Opp: ${aiGoals}  |  Shots: ${shotsLeft}`, 16, 50);

    // instruction hint

    ctx.fillStyle = "rgba(159,179,209,0.8)";

    ctx.fillText(`Stop the pointer in the highlighted zone to score`, 16, h - 18);

    return { goalX, goalY, goalW, goalH, centerX, window };

  }

  function drawPointer(goal) {

    const yTop = goal.goalY + goal.goalH + 18;

    const yBottom = canvas.height * 0.85;

    // guide line

    ctx.strokeStyle = "rgba(231,240,255,0.22)";

    ctx.lineWidth = 2;

    ctx.beginPath();

    ctx.moveTo(pointerX, yTop);

    ctx.lineTo(pointerX, yBottom);

    ctx.stroke();

    // pointer head

    ctx.fillStyle = canShoot ? "rgba(231,240,255,0.95)" : "rgba(231,240,255,0.45)";

    ctx.beginPath();

    ctx.moveTo(pointerX, yTop - 10);

    ctx.lineTo(pointerX - 10, yTop + 10);

    ctx.lineTo(pointerX + 10, yTop + 10);

    ctx.closePath();

    ctx.fill();

    // base marker

    ctx.fillStyle = "rgba(231,240,255,0.55)";

    ctx.fillRect(pointerX - 2, yBottom - 10, 4, 10);

  }

  function tick(t) {

    const dt = Math.min(0.03, (t - lastT) / 1000);

    lastT = t;

    const goal = drawPitch();

    // pointer moves only while fixture running; still moves during pause but canShoot false makes it harder (classic feel)

    if (!fixtureOver) {

      const speed = speedForDivision(division);

      pointerX += pointerDir * speed * dt;

      // bounce inside the goal width (with margins)

      const minX = goal.goalX;

      const maxX = goal.goalX + goal.goalW;

      if (pointerX < minX) { pointerX = minX; pointerDir = 1; }

      if (pointerX > maxX) { pointerX = maxX; pointerDir = -1; }

    }

    drawPointer(goal);

    // If fixture over, show prompt

    if (fixtureOver) {

      ctx.fillStyle = "rgba(0,0,0,0.45)";

      ctx.fillRect(0,0,canvas.width,canvas.height);

      ctx.fillStyle = "rgba(231,240,255,0.95)";

      ctx.font = "bold 26px system-ui, -apple-system, Segoe UI, Roboto, Arial";

      ctx.fillText("Fixture complete", 24, canvas.height/2 - 20);

      ctx.font = "16px system-ui, -apple-system, Segoe UI, Roboto, Arial";

      ctx.fillStyle = "rgba(159,179,209,0.95)";

      ctx.fillText("Tap NEXT FIXTURE (or press N) to continue", 24, canvas.height/2 + 18);

    }

    requestAnimationFrame(tick);

  }

  // init pointer starts mid goal

  pointerX = canvas.width * 0.5;

  updateHUD();

  resetFixture();

  requestAnimationFrame(tick);

})();

</script>
